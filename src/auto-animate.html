						// 牛顿法求解函数
						function newtonMethod(f, fPrime, initialGuess, tolerance, maxIterations) {
								let x = initialGuess;
								let iteration = 0;
								let error = Infinity;
								while (error > tolerance && iteration < maxIterations) {
										let fx = f(x);
										let fxPrime = fPrime(x);
										// 牛顿法迭代公式
										let nextX = x - fx / fxPrime;
										// 计算误差
										error = Math.abs(nextX - x);
										// 更新x的值
										x = nextX;
										iteration++;
								}
								// 如果迭代次数超过最大允许次数，则打印错误信息
								if (iteration === maxIterations) {
										console.log("Newton's method did not converge");
								}
								// 返回近似解
								return x;
						}
						// 示例函数和其导数
						function f(x) { return x * x - 2 }
						function fPrime(x) { return 2 * x }

						// 使用牛顿法求解函数f的根
						let root = newtonMethod(f, fPrime, 1.5, 0.0000001, 100);
